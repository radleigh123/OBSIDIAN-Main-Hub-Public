    public static int linearSearch(ArrayList<Book> books, int searchedId) {
        for (int i = 0; i < books.size(); i++) {
            if (books.get(i).getId() == searchedId) {
                return i;
            }
        }

        return -1;
    }

    public static int binarySearch(ArrayList<Book> books, long searchedId) {
        int start = 0;
        int end = books.size() - 1;
        int middle = 0;

        // perform binary search in this while loop
        while (start <= end) {
            // initialize middle at midpoint of list
            middle = (start + end) / 2;

            // determine next search interval,
            // halving search space each iteration
            Book book = books.get(middle);
            int bookId = book.getId();
            if (bookId == searchedId) {
                return middle;
            } else if (bookId < searchedId) {
                start = middle + 1;
            } else if (bookId > searchedId) {
                end = middle - 1;
            }
        }

        return -1;
    }

    public static int indexOfSmallestFrom(ArrayList<Book> books, int startIndex) {
        int smallestIndex = startIndex;
        int smallest = books.get(startIndex).getId();

        for (int i = startIndex; i < books.size(); i++) {
            if (smallest > books.get(i).getId()) {
                smallest = books.get(i).getId();
                smallestIndex = i;
            }
        }

        return smallestIndex;
    }

    public static void sort(ArrayList<Book> books) {
        int indexOfSmallest = 0;

        for (int i = 0; i < books.size(); i++) {
            indexOfSmallest = indexOfSmallestFrom(books, i);
            swap(books, indexOfSmallest, i);
        }
    }

    public static void swap(ArrayList<Book> books, int index1, int index2) {
        int swap = books.get(index1).getId();
        books.get(index1).setId(books.get(index2).getId());
        books.get(index2).setId(swap);
    }

    public static void print(ArrayList<Book> books) {
        String print = "[";
        for (int i = 0; i < books.size(); i++) {
            print = print + books.get(i).getId();
            print = (i == books.size() - 1) ? print + "]" : print + ", ";
        }

        System.out.println(print);
    }
}

public class Book {
    private int id;
    private String name;

    public Book(int id, String name) {
        this.id = id;
        this.name = name;
    }

    public int getId() {
        return id;
    }

    public String getName() {
        return name;
    }

    @Override
    public String toString() {
        return "(id: " + id + "; name: " + name + ")";
    }
}
