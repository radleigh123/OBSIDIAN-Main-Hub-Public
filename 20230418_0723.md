/**
 * Board class -
 * to display the chess board and show its current view
 */
public class Board {
    char[][] board = {
            { 'K', 'N', 'B', 'R' },
            { 'P', '.', '.', '.' },
            { '.', '.', '.', '.' },
            { '.', '.', '.', 'p' },
            { 'r', 'n', 'b', 'k' }
    };

    private final int boardRw = board.length;
    private final int boardClmn = board[0].length;

    // Display the board and its current position
    public void printBoard() {
        int tmpCount = 1;
        System.out.println(" | a b c d |\n-------------");
        for (int i = 0; i < boardRw; i++) {
            System.out.print(tmpCount + "| ");
            for (int j = 0; j < boardClmn; j++) {
                System.out.print(this.board[i][j] + " ");
            }
            System.out.println("|" + tmpCount++);
        }
        System.out.println("-------------\n | a b c d |");
    }

    // Reset the board to its starting values
    public void resetBoard() {
        char[][] tmpBoard = {
                { 'K', 'N', 'B', 'R' },
                { 'P', '.', '.', '.' },
                { '.', '.', '.', '.' },
                { '.', '.', '.', 'p' },
                { 'r', 'n', 'b', 'k' }
        };

        this.board = tmpBoard;
    }

    int getBoardRw() {
        return boardRw;
    }

    int getBoardClmn() {
        return boardClmn;
    }
}

        setRw(move.charAt(1) - '1');
    }

    boolean isMoveValid() {
        // Input constraints, (a-d) and (1-4) only
        if ((move.charAt(0) >= 'a' && move.charAt(0) <= 'd')
                && (move.charAt(1) >= '1' && move.charAt(1) <= '5')) {
            return true;
        } else {
            return false;
        }
    }

    void setClmn(int clmn) {
        this.clmn = clmn;
    }

    void setRw(int rw) {
        this.rw = rw;
    }

    void setToClmn(int toClmn) {

        this.toRw = toRw;
